Interactive Charts

Powered by Plotly.js for high-performance data visualization.

View Source 
<!-- Include Plotly.js -->
<script src="https://cdn.plot.ly/plotly-2.24.1.min.js"></script>

<!-- Chart Container -->
<div id="chart" class="w-full h-96"></div>

<script>
// Initialize chart
const data = [{
  x: [1, 2, 3, 4, 5],
  y: [1, 4, 9, 16, 25],
  type: 'scatter',
  mode: 'lines+markers',
  line: { color: '#3b82f6' },
  marker: { color: '#2563eb' }
}];

const layout = {
  title: 'Interactive Chart',
  xaxis: { title: 'X Axis' },
  yaxis: { title: 'Y Axis' },
  showlegend: false,
  responsive: true,
  plot_bgcolor: 'rgba(0,0,0,0)',
  paper_bgcolor: 'rgba(0,0,0,0)',
  font: { color: '#1f2937' },
  margin: { t: 30, l: 50, r: 30, b: 50 },
  hovermode: 'closest'
};

// Render chart
Plotly.newPlot('chart', data, layout, { responsive: true });

// Update chart with new data
function updateChart(newData) {
  Plotly.react('chart', newData, layout);
}
</script>

Features

  • Interactive zoom and pan
  • Data point inspection on hover
  • Responsive layout
  • Performance optimized rendering
  • Customizable appearance
