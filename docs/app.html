 157def get_tutorials() -> List[str]:
 158    """Get a list of available OpenFOAM tutorial cases.
 159
 160    Returns:
 161        Sorted list of available OpenFOAM tutorial paths (category/case).
 162    """
 163    try:
 164        client = get_docker_client()
 165        if client is None:
 166            logger.warning("[FOAMFlask] Docker not available, cannot fetch tutorials")
 167            return []
 168
 169        # Get the tutorials directory from OpenFOAM
 170        bashrc = f"/opt/openfoam{OPENFOAM_VERSION}/etc/bashrc"
 171        result = client.containers.run(
 172            DOCKER_IMAGE,
 173            f"bash -c 'source {bashrc} && echo $FOAM_TUTORIALS'",
 174            remove=True,
 175            stdout=True,
 176            stderr=True,
 177            tty=True,
 178        )
 179
 180        tutorial_root = result.decode().strip()
 181        if not tutorial_root:
 182            logger.warning("[FOAMFlask] No tutorial root found in OpenFOAM")
 183            return []
 184
 185        # Find directories containing system/ and constant/ subdirectories
 186        find_cmd = (
 187            f"find {tutorial_root} -mindepth 2 -maxdepth 2 -type d "
 188            "-exec test -d {}/system -a -d {}/constant \\; -print"
 189        )
 190
 191        result = client.containers.run(
 192            DOCKER_IMAGE,
 193            f"bash -c '{find_cmd}'",
 194            remove=True,
 195            stdout=True,
 196            stderr=True,
 197            tty=True,
 198        )
 199
 200        cases = result.decode().splitlines()
 201
 202        # Normalize paths based on OS
 203        if platform.system() == "Windows":
 204            tutorials = [posixpath.relpath(c, tutorial_root) for c in cases]
 205        else:
 206            tutorials = [os.path.relpath(c, tutorial_root) for c in cases]
 207
 208        return sorted(tutorials)
 209
 210    except docker.errors.APIError as e:
 211        logger.error(
 212            "[FOAMFlask] Docker API error while fetching tutorials: %s", str(e)
 213        )
 214    except Exception as e:
 215        logger.error(
 216            "[FOAMFlask] Unexpected error while fetching tutorials: %s",
 217            str(e),
 218            exc_info=True,
 219        )
 220
 221    return []

 228def monitor_foamrun_log(tutorial: str, case_dir: str) -> None:
 229    """Watch for log.FoamRun, capture it in foamrun_logs and write to a file.
 230
 231    Args:
 232        tutorial: The name of the tutorial.
 233        case_dir: The path to the case directory.
 234    """
 235    host_log_path = pathlib.Path(case_dir) / tutorial / "log.FoamRun"
 236    output_file = pathlib.Path(case_dir) / tutorial / "foamrun_logs.txt"
 237
 238    # Configuration
 239    timeout = 300  # seconds max wait
 240    interval = 1  # check every 1 second
 241    elapsed = 0
 242
 243    while elapsed < timeout:
 244        if host_log_path.exists():
 245            try:
 246                log_content = host_log_path.read_text(encoding="utf-8")
 247                foamrun_logs[tutorial] = log_content
 248                output_file.write_text(log_content, encoding="utf-8")
 249                logger.info(
 250                    "[FOAMFlask] Captured log.FoamRun for tutorial '%s' "
 251                    "and wrote to %s",
 252                    tutorial,
 253                    output_file,
 254                )
 255                return
 256            except (OSError, UnicodeDecodeError) as e:
 257                logger.error("[FOAMFlask] Could not process foamrun_logs: %s", str(e))
 258                return
 259
 260        time.sleep(interval)
 261        elapsed += interval
 262
 263    logger.warning("[FOAMFlask] Timeout: log.FoamRun not found for '%s'", tutorial)

 366@app.route("/load_tutorial", methods=["POST"])
 367def load_tutorial():
 368    """
 369    Load a tutorial in the Docker container.
 370
 371    Args:
 372        tutorial (str): The name of the tutorial.
 373
 374    Returns:
 375        dict: The output of the command.
 376    """
 377    global CASE_ROOT, DOCKER_IMAGE, OPENFOAM_VERSION
 378    data = request.get_json()
 379    tutorial = data.get("tutorial")
 380
 381    if not tutorial:
 382        return jsonify({"output": "[FOAMFlask] [Error] No tutorial selected"})
 383
 384    client = get_docker_client()
 385    if client is None:
 386        return docker_unavailable_response()
 387
 388    bashrc = f"/opt/openfoam{OPENFOAM_VERSION}/etc/bashrc"
 389    container_run_path = f"/home/foam/OpenFOAM/{OPENFOAM_VERSION}/run"
 390    container_case_path = posixpath.join(container_run_path, tutorial)
 391
 392    # Convert Windows paths to POSIX style for Docker
 393    host_path = pathlib.Path(CASE_ROOT).resolve()
 394    is_windows = platform.system() == "Windows"
 395    if is_windows:
 396        host_path = host_path.as_posix()  # Docker expects POSIX paths on Windows
 397
 398    # Base docker command: create directory and copy tutorial
 399    docker_cmd = (
 400        f"bash -c 'source {bashrc} && "
 401        f"mkdir -p {container_case_path} && "
 402        f"cp -r $FOAM_TUTORIALS/{tutorial}/* {container_case_path}"
 403    )
 404
 405    # On Linux/macOS, add chmod; on Windows skip it
 406    if not is_windows:
 407        docker_cmd += f" && chmod +x {container_case_path}/Allrun"
 408
 409    docker_cmd += "'"
 410
 411    container = None
 412    try:
 413        container = client.containers.run(
 414            DOCKER_IMAGE,
 415            docker_cmd,
 416            detach=True,
 417            tty=True,
 418            stdout=True,
 419            stderr=True,
 420            volumes={CASE_ROOT: {"bind": container_run_path, "mode": "rw"}},
 421            working_dir=container_run_path,
 422            remove=True,
 423        )
 424
 425        result = container.wait()
 426        logs = container.logs().decode()
 427
 428        if result["StatusCode"] == 0:
 429            output = (
 430                f"INFO::[FOAMFlask] Tutorial loaded::{tutorial}\n"
 431                f"Source: $FOAM_TUTORIALS/{tutorial}\n"
 432                f"Copied to: {CASE_ROOT}/{tutorial}\n"
 433            )
 434        else:
 435            output = f"[FOAMFlask] [Error] Failed to load tutorial {tutorial}\n{logs}"
 436
 437        return jsonify({"output": output, "caseDir": CASE_ROOT})
 438
 439    finally:
 440        if container:
 441            try:
 442                container.reload()
 443                if container.status == "running":
 444                    container.kill()
 445            except Exception:
 446                pass
 447            try:
 448                container.remove()
 449            except Exception:
 450                pass

 453@app.route("/run", methods=["POST"])
 454def run_case():
 455    """
 456    Run a case in the Docker container.
 457
 458    Args:
 459        tutorial (str): The name of the tutorial.
 460        command (str): The command to run.
 461        caseDir (str): The path to the case directory.
 462
 463    Returns:
 464        dict: The output of the command.
 465    """
 466    data = request.json
 467    tutorial = data.get("tutorial")
 468    command = data.get("command")
 469    case_dir = data.get("caseDir")
 470
 471    if not command:
 472        return {"error": "No command provided"}, 400
 473    if not tutorial or not case_dir:
 474        return {"error": "Missing tutorial or caseDir"}, 400
 475
 476    def stream_container_logs():
 477        client = get_docker_client()
 478        if client is None:
 479            # Return a short HTML stream explaining the issue
 480            yield (
 481                "[FOAMFlask] [Error] Docker daemon not available. "
 482                "Please start Docker Desktop and re-run the case.<br>"
 483            )
 484            return
 485
 486        container_case_path = posixpath.join(
 487            f"/home/foam/OpenFOAM/{OPENFOAM_VERSION}/run", tutorial
 488        )
 489        bashrc = f"/opt/openfoam{OPENFOAM_VERSION}/etc/bashrc"
 490
 491        # Convert Windows path to POSIX for Docker volumes
 492        host_path = pathlib.Path(case_dir).resolve().as_posix()
 493        volumes = {
 494            host_path: {
 495                "bind": f"/home/foam/OpenFOAM/{OPENFOAM_VERSION}/run",
 496                "mode": "rw",
 497            }
 498        }
 499
 500        # Start the watcher thread before running container
 501        watcher_thread = threading.Thread(
 502            target=monitor_foamrun_log, args=(tutorial, case_dir), daemon=True
 503        )
 504        watcher_thread.start()
 505
 506        # Determine if command is an OpenFOAM command or a script file
 507        openfoam_commands = ["blockMesh", "simpleFoam", "pimpleFoam", "decomposePar", "reconstructPar", "foamToVTK", "paraFoam"]
 508        
 509        if command.startswith("./") or command in openfoam_commands:
 510            if command.startswith("./"):
 511                # Script file - make executable and run
 512                docker_cmd = f"bash -c 'source {bashrc} && cd {container_case_path} && chmod +x {command} && {command}'"
 513            else:
 514                # OpenFOAM command - run directly after sourcing bashrc
 515                docker_cmd = f"bash -c 'source {bashrc} && cd {container_case_path} && {command}'"
 516        else:
 517            # Fallback to original behavior
 518            docker_cmd = f"bash -c 'source {bashrc} && cd {container_case_path} && chmod +x {command} && ./{command}'"
 519
 520        container = client.containers.run(
 521            DOCKER_IMAGE,
 522            docker_cmd,
 523            detach=True,
 524            tty=False,
 525            volumes=volumes,
 526            working_dir=container_case_path,
 527        )
 528
 529        try:
 530            # Stream logs line by line
 531            for line in container.logs(stream=True):
 532                decoded = line.decode(errors="ignore")
 533                for subline in decoded.splitlines():
 534                    yield subline + "<br>"
 535
 536        finally:
 537            try:
 538                container.kill()
 539            except:
 540                pass
 541            try:
 542                container.remove()
 543            except:
 544                logger.error("[FOAMPilot] Could not remove container")
 545
 546    return Response(stream_container_logs(), mimetype="text/html")
